<script>
    let resources = JSON.parse(localStorage.getItem('hy_resources_v10_db')) || { ideas: [] };

    window.addSmartIdea = function() {
        let input = document.getElementById('input-idea');
        let url = input.value.trim();
        if(!url) return;

        let platform = "Khác", username = "Chưa rõ";
        try {
            let u = new URL(url);
            let host = u.hostname.toLowerCase();
            let path = u.pathname.split('/');
            
            if (host.includes('x.com') || host.includes('twitter.com')) { platform = "X (Twitter)"; username = path[1] || "user"; }
            else if (host.includes('threads.net')) { platform = "Threads"; username = path[1] || "user"; }
            else if (host.includes('facebook.com')) { platform = "Facebook"; username = path[1] || "user"; }
            else if (host.includes('instagram.com')) { platform = "Instagram"; username = path[1] || "user"; }
            else if (host.includes('tiktok.com')) { platform = "TikTok"; username = (path[1] || "user").replace('@', ''); }
            else if (host.includes('xiaohongshu.com')) { platform = "Xiaohongshu"; username = "XHS_User"; }
        } catch(e) { console.log("Link lỗi"); }

        resources.ideas.push({
            id: Date.now(), 
            link: url, 
            title: '[Gõ Tiêu đề]', 
            platform: platform, 
            username: username, 
            cre: `Cre: ${platform}, @${username}`, 
            checked: false
        });
        
        localStorage.setItem('hy_resources_v10_db', JSON.stringify(resources));
        input.value = ''; 
        renderResources();
    }

    function renderResources() {
        let tbody = document.getElementById('list-ideas'); tbody.innerHTML = '';
        resources.ideas.forEach(item => {
            let tr = document.createElement('tr');
            tr.innerHTML = `
                <td style="text-align:center;"><input type="checkbox" onchange="toggleIdeaCheck(${item.id})" ${item.checked ? 'checked' : ''}></td>
                <td><a href="${item.link}" target="_blank">🔗 Mở</a></td>
                <td class="editable-cell" contenteditable="true" onblur="updateIdeaField(${item.id}, 'title', this.innerText)">${item.title}</td>
                <td class="editable-cell" contenteditable="true" onblur="updateIdeaField(${item.id}, 'platform', this.innerText)">${item.platform}</td>
                <td class="editable-cell" contenteditable="true" onblur="updateIdeaField(${item.id}, 'username', this.innerText)">${item.username}</td>
                <td class="editable-cell" contenteditable="true" onblur="updateIdeaField(${item.id}, 'cre', this.innerText)">${item.cre}</td>
                <td style="text-align:center;"><button class="delete-custom-btn" onclick="deleteResource(${item.id})">×</button></td>
            `;
            tbody.appendChild(tr);
        });
    }

    window.updateIdeaField = function(id, field, value) {
        let target = resources.ideas.find(i => i.id === id);
        if(target) { target[field] = value.trim(); localStorage.setItem('hy_resources_v10_db', JSON.stringify(resources)); }
    }

    window.toggleIdeaCheck = function(id) {
        let target = resources.ideas.find(r => r.id === id);
        if(target) { target.checked = !target.checked; localStorage.setItem('hy_resources_v10_db', JSON.stringify(resources)); }
    }

    window.deleteResource = function(id) {
        resources.ideas = resources.ideas.filter(i => i.id !== id);
        localStorage.setItem('hy_resources_v10_db', JSON.stringify(resources));
        renderResources();
    }

    function switchView(view) {
        document.getElementById('resources-box').style.display = (view === 'resources') ? 'block' : 'none';
        renderResources();
    }
</script>
